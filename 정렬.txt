#include "stdafx.h"
#include "stdlib.h"

int _tmain(int argc, _TCHAR* argv[])
{
int arr[10];
int change;
int i,j;

printf("정수 입력 : ");
for(i=0;i<10;i++)
scanf("%d",&arr[i]);
printf("정렬전 배열 : ");
for(i=0;i<10;i++)
printf("%d ",arr[i]);
printf("\n");

for(i=0; i < 10; i++) 
for(j=10-2; j > i-1; --j)
{
if(arr[j+1]<arr[j])
{
change=arr[j];
arr[j]=arr[j+1];
arr[j+1]=change;
}
}

printf("정렬후 배열 : ");
for(i=0;i<10;i++)
printf("%d ",arr[i]);
printf("\n");




system("pause");
return 0;
}
