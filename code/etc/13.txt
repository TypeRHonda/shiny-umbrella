

#include "stdio.h"



int main ()
{
 int A[3][2] = {{1,4},{2,5},{3,6}};
 int B[2][3] = {{7,8,9},{10,11,12}};
 int C[3][3];
 int i,j;

 for(i=0;i<3;i++)
 {
	 for(j=0;j<3;j++)
	 {
			  C[i][j] = A[i][0]*B[0][j]+A[i][1]*B[1][j];
	 }
 }
 printf("실행 결과 \n");
 for(i=0;i<3;i++)
 {
	 for(j=0;j<3;j++)
	 {
		 printf("%2d ",C[i][j]);
	 }
	 printf("\n");
 }
 return 0;
}
