#include "stdio.h"
#include "stdlib.h"


int main (void)
{
char ch;
double num;
double fah,cel;
printf("섭씨면 'C', 화씨면 'F'를 입력하세요: ");
scanf("%c",&ch);


if(ch=='c'|| ch== 'C')
{
printf("섭씨 온도를 입력하세요 :");
scanf("%lf",&num);

fah = num*1.8+32;
printf("화씨 온도는 %f 입니다.\n",fah);

}
else if (ch=='f'|| ch=='F')
{
printf("화씨 온도를 입력하세요 : ");
scanf("%lf",&num);

cel = (num-32)/1.8;
printf("섭씨 온도는 %f 입니다.\n",cel);
}



system ("pause");

return 0 ;
}
